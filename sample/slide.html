<1. 단순> 자동 + 수평

$('.slide1 > .slide-screen').jui('slide');

<2. 일반> 자동 + 수평 + 컨트롤

$('.slide2 > .slide-screen').jui('slide', {
	slideInterval : 3000,	// 슬라이드 시간 간격 (3sec)
	prevButton    : $('.slide2 .slide-prev'),   // 이전버튼           
	nextButton    : $('.slide2 .slide-next'),   // 다음버튼           
	toggleButton  : $('.slide2 .slide-toggle'), // 토글(동작/정지)버튼
	play          : function (settings) {
		settings.toggleButton.html('■');
	},
	stop          : function (settings) {
		settings.toggleButton.html('▶');
	}
});

<3. 고급> 수동 + 수평 + 컨트롤 + 네비게이션

$('.slide3 > .slide-screen').jui('slide', {
	slideAuto       : false, // 수동 슬라이드
	slideInterval   : 2000,  // 슬라이드 시간 간격 (2sec)
	prevButton      : $('.slide3 .slide-prev'), // 이전버튼
	nextButton      : $('.slide3 .slide-next'), // 다음버튼
	playButton      : $('.slide3 .slide-play'), // 동작버튼
	stopButton      : $('.slide3 .slide-stop'), // 정지버튼
	slideNavigation : $('.slide3 .slide-navigation button'),
	play            : function (settings) {
		settings.playButton.attr('disabled', true);
		settings.stopButton.attr('disabled', false);
		settings.stopButton.get(0).focus();
	},
	stop            : function (settings) {
		settings.playButton.attr('disabled', false);
		settings.stopButton.attr('disabled', true);
		settings.playButton.get(0).focus();
	}
});

<고급> 수동 + 수직 + 속도

$('.slide4 > .slide-screen').jui('slide', {
	slideDirection : 'vertical', // 수직 슬라이드
	slideAuto      : false,      // 수동 슬라이드
	slideInterval  : 2500,       // 슬라이드 시간 간격 (2.5sec)
	slideSpeed     : 'slow',     // 슬라이드 애니메이션 속도
	prevButton     : $('.slide4 .slide-prev'), // 이전버튼
	nextButton     : $('.slide4 .slide-next'), // 다음버튼
	playButton     : $('.slide4 .slide-play'), // 동작버튼
	stopButton     : $('.slide4 .slide-stop'), // 정지버튼
	play           : function (settings) {
		settings.playButton.attr('disabled', true);
		settings.stopButton.attr('disabled', false);
		settings.stopButton.get(0).focus();
	},
	stop			: function (settings) {
		settings.playButton.attr('disabled', false);
		settings.stopButton.attr('disabled', true);
		settings.playButton.get(0).focus();
	}
});

prevButton ㅇ $()
